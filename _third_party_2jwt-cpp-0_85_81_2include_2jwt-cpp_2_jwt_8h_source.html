366 inline std::string extract_pubkey_from_cert(const std::string& certstr, const std::string& pw,
367 std::error_code& ec) {
368 ec.clear();
369#if OPENSSL_VERSION_NUMBER <= 0x10100003L
370 std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(
371 BIO_new_mem_buf(const_cast<char*>(certstr.data()), static_cast<int>(certstr.size())), BIO_free_all);
372#else
373 std::unique_ptr<BIO, decltype(&BIO_free_all)> certbio(
374 BIO_new_mem_buf(certstr.data(), static_cast<int>(certstr.size())), BIO_free_all);
375#endif
376 std::unique_ptr<BIO, decltype(&BIO_free_all)> keybio(BIO_new(BIO_s_mem()), BIO_free_all);
377 if (!certbio || !keybio) {
378 ec = error::rsa_error::create_mem_bio_failed;
379 return {};
380 }
381
382 std::unique_ptr<X509, decltype(&X509_free)> cert(
383 PEM_read_bio_X509(certbio.get(), nullptr, nullptr, const_cast<char*>(pw.c_str())), X509_free);
384 if (!cert) {
385 ec = error::rsa_error::cert_load_failed;
386 return {};
387 }
388 std::unique_ptr<EVP_PKEY, decltype(&EVP_PKEY_free)> key(X509_get_pubkey(cert.get()), EVP_PKEY_free);
389 if (!key) {
390 ec = error::rsa_error::get_key_failed;
391 return {};
392 }
393 if (PEM_write_bio_PUBKEY(keybio.get(), key.get()) == 0) {
394 ec = error::rsa_error::write_key_failed;
395 return {};
396 }
397 char* ptr = nullptr;
398 auto len = BIO_get_mem_data(keybio.get(), &ptr);
399 if (len <= 0 || ptr == nullptr) {
400 ec = error::rsa_error::convert_to_pem_failed;
401 return {};
402 }
403 return {ptr, static_cast<size_t>(len)};
404 }

527 inline std::shared_ptr<EVP_PKEY> load_public_key_from_string(const std::string& key,
528 const std::string& password, std::error_code& ec) {
529 ec.clear();
530 std::unique_ptr<BIO, decltype(&BIO_free_all)> pubkey_bio(BIO_new(BIO_s_mem()), BIO_free_all);
531 if (!pubkey_bio) {
532 ec = error::rsa_error::create_mem_bio_failed;
533 return nullptr;
534 }
535 if (key.substr(0, 27) == "-----BEGIN CERTIFICATE-----") {
536 auto epkey = helper::extract_pubkey_from_cert(key, password, ec);
537 if (ec) return nullptr;
538 const int len = static_cast<int>(epkey.size());
539 if (BIO_write(pubkey_bio.get(), epkey.data(), len) != len) {
540 ec = error::rsa_error::load_key_bio_write;
541 return nullptr;
542 }
543 } else {
544 const int len = static_cast<int>(key.size());
545 if (BIO_write(pubkey_bio.get(), key.data(), len) != len) {
546 ec = error::rsa_error::load_key_bio_write;
547 return nullptr;
548 }
549 }
550
551 std::shared_ptr<EVP_PKEY> pkey(
552 PEM_read_bio_PUBKEY(pubkey_bio.get(), nullptr, nullptr,
553 (void*)password.data()), // NOLINT(google-readability-casting) requires `const_cast`
554 EVP_PKEY_free);
555 if (!pkey) {
556 ec = error::rsa_error::load_key_bio_read;
557 return nullptr;
558 }
559 return pkey;
560 }

700 hmacsha(std::string key, const EVP_MD* (*md)(), std::string name)
701 : secret(std::move(key)), md(md), alg_name(std::move(name)) {}
708 std::string sign(const std::string& data, std::error_code& ec) const {
709 ec.clear();
710 std::string res(static_cast<size_t>(EVP_MAX_MD_SIZE), '\0');
711 auto len = static_cast<unsigned int>(res.size());
712 if (HMAC(md(), secret.data(), static_cast<int>(secret.size()),
713 reinterpret_cast<const unsigned char*>(data.data()), static_cast<int>(data.size()),
714 (unsigned char*)res.data(), // NOLINT(google-readability-casting) requires `const_cast`
715 &len) == nullptr) {
716 ec = error::signature_generation_error::hmac_failed;
717 return {};
718 }
719 res.resize(len);
720 return res;
721 }
728 void verify(const std::string& data, const std::string& signature, std::error_code& ec) const {
729 ec.clear();
730 auto res = sign(data, ec);
731 if (ec) return;
732
733 bool matched = true;
734 for (size_t i = 0; i < std::min<size_t>(res.size(), signature.size()); i++)
735 if (res[i] != signature[i]) matched = false;
736 if (res.size() != signature.size()) matched = false;
737 if (!matched) {
738 ec = error::signature_verification_error::invalid_signature;
739 return;
740 }
741 }
746 std::string name() const { return alg_name; }
747
748 private:
750 const std::string secret;
752 const EVP_MD* (*md)();
754 const std::string alg_name;

759 struct rsa {
769 rsa(const std::string& public_key, const std::string& private_key, const std::string& public_key_password,
770 const std::string& private_key_password, const EVP_MD* (*md)(), std::string name)

785 std::string sign(const std::string& data, std::error_code& ec) const {
786 ec.clear();
787#ifdef OPENSSL10
788 std::unique_ptr<EVP_MD_CTX, decltype(&EVP_MD_CTX_destroy)> ctx(EVP_MD_CTX_create(), EVP_MD_CTX_destroy);
789#else
790 std::unique_ptr<EVP_MD_CTX, decltype(&EVP_MD_CTX_free)> ctx(EVP_MD_CTX_create(), EVP_MD_CTX_free);
791#endif
792 if (!ctx) {
793 ec = error::signature_generation_error::create_context_failed;
794 return {};
795 }
796 if (!EVP_SignInit(ctx.get(), md())) {
797 ec = error::signature_generation_error::signinit_failed;
798 return {};
799 }
800
801 std::string res(EVP_PKEY_size(pkey.get()), '\0');
802 unsigned int len = 0;
803
804 if (!EVP_SignUpdate(ctx.get(), data.data(), data.size())) {
805 ec = error::signature_generation_error::signupdate_failed;
806 return {};
807 }
808 if (EVP_SignFinal(ctx.get(), (unsigned char*)res.data(), &len, pkey.get()) == 0) {
809 ec = error::signature_generation_error::signfinal_failed;
810 return {};
811 }
812
813 res.resize(len);
814 return res;
815 }
822 void verify(const std::string& data, const std::string& signature, std::error_code& ec) const {
823 ec.clear();
824#ifdef OPENSSL10
825 std::unique_ptr<EVP_MD_CTX, decltype(&EVP_MD_CTX_destroy)> ctx(EVP_MD_CTX_create(), EVP_MD_CTX_destroy);
826#else
827 std::unique_ptr<EVP_MD_CTX, decltype(&EVP_MD_CTX_free)> ctx(EVP_MD_CTX_create(), EVP_MD_CTX_free);
828#endif
829 if (!ctx) {
830 ec = error::signature_verification_error::create_context_failed;
831 return;
832 }
833 if (!EVP_VerifyInit(ctx.get(), md())) {
834 ec = error::signature_verification_error::verifyinit_failed;
835 return;
836 }
837 if (!EVP_VerifyUpdate(ctx.get(), data.data(), data.size())) {
